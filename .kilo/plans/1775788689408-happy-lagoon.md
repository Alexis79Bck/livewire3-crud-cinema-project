### Mappers

**AuditoriumMapper.php:**
```php
class AuditoriumMapper
{
    public static function toDomain(Auditorium $model): Auditorium
    {
        $seats = $model->seats->map(fn($seat) => SeatMapper::toDomain($seat))->toArray();
        
        return Auditorium::reconstitute(
            new AuditoriumId($model->id),
            $model->name,
            $model->capacity,
            $model->location,
            AuditoriumStatus::from($model->status),
            $seats
        );
    }
    
    public static function toEloquent(Auditorium $auditorium): Auditorium
    {
        $model = Auditorium::find($auditorium->id()->value()) ?? new Auditorium();
        
        $model->id = $auditorium->id()->value();
        $model->name = $auditorium->name();
        $model->capacity = $auditorium->capacity();
        $model->location = $auditorium->location();
        $model->status = $auditorium->status()->value;
        
        return $model;
    }
}
```

**SeatMapper.php:**
```php
class SeatMapper
{
    public static function toDomain(Seat $model): Seat
    {
        return Seat::create(
            new SeatNumber($model->row, $model->seat_number),
            SeatType::from($model->type)
        );
    }
    
    public static function toEloquent(Seat $seat, string $auditoriumId): Seat
    {
        $model = new Seat();
        $model->id = Uuid::uuid4()->toString();
        $model->auditorium_id = $auditoriumId;
        $model->row = $seat->seatNumber()->row();
        $model->seat_number = $seat->seatNumber()->number();
        $model->type = $seat->type()->value;
        $model->is_available = true;
        
        return $model;
    }
}
```

### Repositorio Eloquent

**EloquentAuditoriumRepository.php:**
```php
class EloquentAuditoriumRepository implements AuditoriumRepository
{
    public function save(Auditorium $auditorium): void
    {
        DB::transaction(function () use ($auditorium) {
            $model = AuditoriumMapper::toEloquent($auditorium);
            $model->save();
            
            // Handle seats
            Seat::where('auditorium_id', $auditorium->id()->value())->delete();
            
            foreach ($auditorium->seats() as $seat) {
                $seatModel = SeatMapper::toEloquent($seat, $auditorium->id()->value());
                $seatModel->save();
            }
        });
    }
    
    public function findById(AuditoriumId $id): ?Auditorium
    {
        $model = Auditorium::with('seats')->find($id->value());
        
        if (!$model) {
            return null;
        }
        
        return AuditoriumMapper::toDomain($model);
    }
    
    public function delete(Auditorium $auditorium): void
    {
        Auditorium::where('id', $auditorium->id()->value())->delete();
    }
    
    public function findAll(): array
    {
        $models = Auditorium::with('seats')->get();
        
        return $models->map(fn($model) => AuditoriumMapper::toDomain($model))->toArray();
    }
    
    public function findByStatus(AuditoriumStatus $status): array
    {
        $models = Auditorium::with('seats')->where('status', $status->value)->get();
        
        return $models->map(fn($model) => AuditoriumMapper::toDomain($model))->toArray();
    }
}
```
